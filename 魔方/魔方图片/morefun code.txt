        // 轮播功能
        document.addEventListener('DOMContentLoaded', function() {
            const slides = document.querySelectorAll('.slide');
            const thumbnails = document.querySelectorAll('.thumbnail');
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            let currentIndex = 0;
            let slideInterval;
            
            // 初始化轮播
            function initSlider() {
                // 隐藏所有幻灯片，只显示当前活动的
                slides.forEach((slide, index) => {
                    if (index === currentIndex) {
                        slide.classList.add('active');
                        slide.style.display = 'block';
                    } else {
                        slide.classList.remove('active');
                        slide.style.display = 'none';
                    }
                });
                
                // 激活当前缩略图
                thumbnails.forEach((thumbnail, index) => {
                    if (index === currentIndex) {
                        thumbnail.classList.add('active');
                    } else {
                        thumbnail.classList.remove('active');
                    }
                });
                
                startAutoSlide();
            }
            
            // 显示指定幻灯片
            function showSlide(index) {
                // 隐藏所有幻灯片
                slides.forEach(slide => {
                    slide.classList.remove('active');
                    slide.style.display = 'none';
                });
                
                // 移除所有缩略图的active类
                thumbnails.forEach(thumbnail => {
                    thumbnail.classList.remove('active');
                });
                
                // 显示当前幻灯片
                slides[index].classList.add('active');
                slides[index].style.display = 'block';
                // 激活当前缩略图
                thumbnails[index].classList.add('active');
                
                // 更新当前索引
                currentIndex = index;
            }
            
            // 下一张幻灯片
            function nextSlide() {
                let nextIndex = currentIndex + 1;
                if (nextIndex >= slides.length) {
                    nextIndex = 0;
                }
                showSlide(nextIndex);
                resetAutoSlide();
            }
            
            // 上一张幻灯片
            function prevSlide() {
                let prevIndex = currentIndex - 1;
                if (prevIndex < 0) {
                    prevIndex = slides.length - 1;
                }
                showSlide(prevIndex);
                resetAutoSlide();
            }
            
            // 自动轮播
            function startAutoSlide() {
                slideInterval = setInterval(nextSlide, 5000); // 每5秒切换一次
            }
            
            // 重置自动轮播
            function resetAutoSlide() {
                clearInterval(slideInterval);
                startAutoSlide();
            }
            
            // 事件监听
            nextBtn.addEventListener('click', nextSlide);
            prevBtn.addEventListener('click', prevSlide);
            
            // 点击缩略图切换幻灯片
            thumbnails.forEach((thumbnail, index) => {
                thumbnail.addEventListener('click', () => {
                    showSlide(index);
                    resetAutoSlide();
                });
            });
            
            // 初始化
            initSlider();
        });

                const containerWidth = gamesWrapper.offsetWidth;
                const centerPosition = containerWidth / 2;
                
                // 计算所有游戏项的位置
                let totalWidth = 0;
                for (let i = 0; i < currentIndex; i++) {
                    totalWidth += gameItems[i].offsetWidth + 20; // 20是间距
                }
                
                // 计算当前项的中心位置
                const currentItemWidth = gameItems[currentIndex].offsetWidth;
                const currentItemCenter = totalWidth + currentItemWidth / 2;
                
                // 计算偏移量
                const offset = centerPosition - currentItemCenter;
                gamesGrid.style.transform = `translateX(${offset}px)`;
            }
            
            // 滚动到下一个游戏
            function scrollToNext() {
                // 移除当前项的激活状态
                gameItems[currentIndex].classList.remove('active');
                
                // 增加索引
                currentIndex++;
                
                // 当滚动到复制的游戏项时，无缝切换到原始游戏项
                if (currentIndex >= itemCount) {
                    // 瞬间切换到对应的原始游戏项
                    currentIndex = currentIndex - itemCount;
                    
                    // 禁用过渡效果
                    gamesGrid.style.transition = 'none';
                    
                    // 更新位置
                    updateGridPosition();
                    
                    // 强制重排
                    void gamesGrid.offsetWidth;
                    
                    // 恢复过渡效果
                    gamesGrid.style.transition = 'transform 0.5s ease';
                }
                
                // 添加新项的激活状态
                gameItems[currentIndex].classList.add('active');
                
                // 更新位置
                updateGridPosition();
            }
            
            // 滚动到上一个游戏
            function scrollToPrev() {
                // 移除当前项的激活状态
                gameItems[currentIndex].classList.remove('active');
                
                // 减少索引
                currentIndex--;
                
                // 当滚动到原始游戏项的开头时，无缝切换到复制的游戏项
                if (currentIndex < 0) {
                    // 瞬间切换到对应的复制游戏项
                    currentIndex = currentIndex + itemCount;
                    
                    // 禁用过渡效果
                    gamesGrid.style.transition = 'none';
                    
                    // 更新位置
                    updateGridPosition();
                    
                    // 强制重排
                    void gamesGrid.offsetWidth;
                    
                    // 恢复过渡效果
                    gamesGrid.style.transition = 'transform 0.5s ease';
                }
                
                // 添加新项的激活状态
                gameItems[currentIndex].classList.add('active');
                
                // 更新位置
                updateGridPosition();
            }
            
            // 点击游戏项
            gamesGrid.addEventListener('click', function(e) {
                const gameItem = e.target.closest('.game-item');
                if (gameItem) {
                    scrollToNext();
                }
            });
            
            // 左箭头点击
            leftArrow.addEventListener('click', function() {
                scrollToPrev();
            });
            
            // 右箭头点击
            rightArrow.addEventListener('click', function() {
                scrollToNext();
            });
            
            // 窗口大小变化时重新计算位置
            window.addEventListener('resize', updateGridPosition);
            
            // 初始化
            initGames();
        });
    </script>
</body>
